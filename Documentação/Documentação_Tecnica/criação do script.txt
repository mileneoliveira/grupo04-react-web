-- criação da tabela para luiza e rapha

CREATE TABLE Alimento(
	idAlimento int NOT NULL IDENTITY(1,1) PRIMARY KEY,
	nome varchar(45) NOT NULL,
	porcao double NOT NULL,
	proteina double NOT NULL,
	colesterol double NOT NULL,
	carboidrato double NOT NULL,
	fibra double NOT NULL,
	calcio double NOT NULL,
	ferro double NOT NULL,
	sodio double NOT NULL,
	calorias double NOT NULL,
	categoriaId int NOT NULL,
	FOREIGN KEY (idCategoria) REFERENCES Categoria(categoriaId)
);
